Sub PadRightToSixDigits()
    Dim colNum As Integer
    Dim lastRow As Long
    Dim i As Long
    Dim val As String
    Dim newVal As String
    
    colNum = InputBox("D", , 4)
    If colNum = 0 Then Exit Sub
    
    lastRow = ActiveSheet.UsedRange.Rows.Count
    Application.ScreenUpdating = False
    
    For i = 1 To lastRow
        If Not IsError(Cells(i, colNum).Value) Then
            val = Trim(CStr(Cells(i, colNum).Value))
            If Len(val) < 6 Then
                newVal = val & String(6 - Len(val), "0")
                Cells(i, colNum).Value = newVal
            End If
        End If
    Next i
    
    Application.ScreenUpdating = True
    MsgBox "Done"
End Sub
