Sub RemoveCommas()
    Dim colNum As Integer
    Dim lastRow As Long
    Dim i As Long
    
    colNum = InputBox("D", , 4)
    If colNum = 0 Then Exit Sub
    
    lastRow = ActiveSheet.UsedRange.Rows.Count
    Application.ScreenUpdating = False
    
    For i = 1 To lastRow
        If Not IsError(Cells(i, colNum).Value) Then
            Cells(i, colNum).Value = Replace(Cells(i, colNum).Value, ",", "")
        End If
    Next i
    
    Application.ScreenUpdating = True
    MsgBox "done"
End Sub
